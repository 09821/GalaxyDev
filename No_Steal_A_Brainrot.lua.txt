-- Lista

list.CanvasSize = UDim2.new(0,0,0,0)
list.AutomaticCanvasSize = Enum.AutomaticSize.Y
list.ScrollBarImageColor3 = THEME_COLOR

local layout = Instance.new("UIListLayout", list)
layout.Padding = UDim.new(0,6)

local selectedModel = nil
local spawnCFrame = HRP.CFrame

-- Função para ler Brainrots
local function loadBrainrots()
	list:ClearAllChildren()
	layout.Parent = list

	local folder = workspace:FindFirstChild("ActiveBrainrots")
	if not folder then return end

	local entries = {}

	for _,rarity in ipairs(folder:GetChildren()) do
		if rarity:IsA("Folder") then
			for _,model in ipairs(rarity:GetChildren()) do
				if model:IsA("Model") then
					table.insert(entries, {
						model = model,
						text = model.Name .. " (" .. rarity.Name .. ")"
					})
				end
			end
		end
	end

	table.sort(entries, function(a,b)
		return a.text:lower() < b.text:lower()
	end)

	for _,info in ipairs(entries) do
		local b = Instance.new("TextButton", list)
		b.Size = UDim2.new(1,-10,0,35)
		b.Text = info.text
		b.BackgroundColor3 = Color3.fromRGB(50,40,90)
		b.TextColor3 = Color3.new(1,1,1)
		b.Font = Enum.Font.Gotham
		b.TextSize = 14
		Instance.new("UICorner", b)

		b.MouseButton1Click:Connect(function()
			selectedModel = info.model
			main.Visible = false
		end)
	end
end

-- Toggle GUI
circle.MouseButton1Click:Connect(function()
	main.Visible = not main.Visible
	if main.Visible then
		loadBrainrots()
	end
end)

-- Catch Brainrot
game:GetService("UserInputService").InputBegan:Connect(function(i,g)
	if g then return end
	if i.KeyCode == Enum.KeyCode.C and selectedModel then
		if selectedModel.PrimaryPart then
			selectedModel:SetPrimaryPartCFrame(HRP.CFrame)
		end
	end
end)
